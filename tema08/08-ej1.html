Ejercicio 1: Analizador de Sentimientos con BERT (Nivel: Principiante)

🎯 Objetivo

Crear un clasificador que determine si una reseña de película es positiva o negativa usando BERT pre-entrenado.

📝 Descripción

Usa BERT ya entrenado y solo entrena una capa de clasificación encima. Aprenderás:

  • Cómo cargar modelos pre-entrenados
  • Fine-tuning básico
  • Preparación de datos para transformers

💻 Código Inicial

import torch
import torch.nn as nn
from torch.utils.data import Dataset, DataLoader
from transformers import BertTokenizer, BertModel
from sklearn.model_selection import train_test_split
import numpy as np

# === PASO 1: Dataset ===
class SentimentDataset(Dataset):
    """Dataset para análisis de sentimientos"""
    def __init__(self, texts, labels, tokenizer, max_len=128):
        self.texts = texts
        self.labels = labels
        self.tokenizer = tokenizer
        self.max_len = max_len
    
    def __len__(self):
        return len(self.texts)
    
    def __getitem__(self, idx):
        text = str(self.texts[idx])
        label = self.labels[idx]
        
        # Tokenizar con BERT tokenizer
        encoding = self.tokenizer.encode_plus(
            text,
            add_special_tokens=True,
            max_length=self.max_len,
            padding='max_length',
            truncation=True,
            return_attention_mask=True,
            return_tensors='pt'
        )
        
        return {
            'input_ids': encoding['input_ids'].flatten(),
            'attention_mask': encoding['attention_mask'].flatten(),
            'label': torch.tensor(label, dtype=torch.long)
        }


# === PASO 2: Modelo ===
class SentimentClassifier(nn.Module):
    def __init__(self, n_classes=2):
        super().__init__()
        
        # BERT pre-entrenado
        self.bert = BertModel.from_pretrained('bert-base-uncased')
        
        # Congelar parámetros de BERT (opcional, hace entrenamiento más rápido)
        # Descomenta la siguiente línea si quieres solo entrenar el clasificador
        # for param in self.bert.parameters():
        #     param.requires_grad = False
        
        # Capa de clasificación
        self.dropout = nn.Dropout(0.3)
        self.classifier = nn.Linear(self.bert.config.hidden_size, n_classes)
    
    def forward(self, input_ids, attention_mask):
        # Pasar por BERT
        outputs = self.bert(
            input_ids=input_ids,
            attention_mask=attention_mask
        )
        
        # Usar el output del token [CLS]
        pooled_output = outputs.pooler_output
        
        # Dropout y clasificación
        output = self.dropout(pooled_output)
        logits = self.classifier(output)
        
        return logits

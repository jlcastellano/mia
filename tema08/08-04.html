🎓 Analogía: Enseñar a un Estudiante a Traducir

Fase 1: Mostrar ejemplos
Le das al estudiante miles de pares de oraciones:

"Hello" → "Hola"

"How are you?" → "¿Cómo estás?"

Fase 2: Práctica con pistas

Pistas ya dadas: "El gato"

Si el estudiante dice "come" en lugar de "duerme", le corriges y él ajusta su comprensión.

Este ciclo se repite millones de veces hasta que aprende.

4.2 Código de Entrenamiento Completo

import torch
import torch.nn as nn
import torch.optim as optim
from torch.utils.data import Dataset, DataLoader

# === DATASET ===
class TranslationDataset(Dataset):
    """
    Dataset simple para traducción.
    En la práctica, usarías datasets como Multi30k o WMT.
    """
    def __init__(self, pairs, src_vocab, tgt_vocab, max_len=50):
        """
        pairs: Lista de tuplas (inglés, español)
        src_vocab: Diccionario {palabra: id} para inglés
        tgt_vocab: Diccionario {palabra: id} para español
        """
        self.pairs = pairs
        self.src_vocab = src_vocab
        self.tgt_vocab = tgt_vocab
        self.max_len = max_len
    
    def text_to_ids(self, text, vocab):
        """Convertir texto a lista de IDs"""
        words = text.lower().split()
        ids = [vocab.get(word, vocab['<UNK>']) for word in words]
        # Añadir tokens especiales
        ids = [vocab['<SOS>']] + ids + [vocab['<EOS>']]
        # Padding
        if len(ids) < self.max_len:
            ids = ids + [vocab['<PAD>']] * (self.max_len - len(ids))
        else:
            ids = ids[:self.max_len]
        return ids
    
    def __len__(self):
        return len(self.pairs)
    
    def __getitem__(self, idx):
        src_text, tgt_text = self.pairs[idx]
        
        src_ids = self.text_to_ids(src_text, self.src_vocab)
        tgt_ids = self.text_to_ids(tgt_text, self.tgt_vocab)
        
        return torch.tensor(src_ids), torch.tensor(tgt_ids)


# === FUNCIONES DE ENTRENAMIENTO ===

def train_step(model, src, tgt, optimizer, criterion, device):
    """Un paso de entrenamiento"""
    model.train()
    
    src = src.to(device)
    tgt = tgt.to(device)
    
    # El decoder input es todo excepto la última palabra
    tgt_input = tgt[:, :-1]
    # Los labels son todo excepto la primera palabra (desplazado)
    tgt_labels = tgt[:, 1:]
    
    # Forward pass
    optimizer.zero_grad()
    logits = model(src, tgt_input)
    
    # Calcular pérdida
    # logits: [batch, seq_len-1, vocab_size]
    # tgt_labels: [batch, seq_len-1]
    loss = criterion(
        logits.reshape(-1, logits.size(-1)),
        tgt_labels.reshape(-1)
    )
    
    # Backward pass
    loss.backward()
    
    # Gradient clipping (evitar gradientes explosivos)
    torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=1.0)
    
    optimizer.step()
    
    return loss.item()


def train_epoch(model, dataloader, optimizer, criterion, device):
    """Entrenar una época completa"""
    total_loss = 0
    num_batches = len(dataloader)
    
    for batch_idx, (src, tgt) in enumerate(dataloader):
        loss = train_step(model, src, tgt, optimizer, criterion, device)
        total_loss += loss
        
        if (batch_idx + 1) % 50 == 0:
            avg_loss = total_loss / (batch_idx + 1)
            print(f"  Batch {batch_idx + 1}/{num_batches} - Loss: {avg_loss:.4f}")
    
    return total_loss / num_batches


@torch.no_grad()
def evaluate(model, dataloader, criterion, device):
    """Evaluar el modelo"""
    model.eval()
    total_loss = 0
    
    for src, tgt in dataloader:
        src = src.to(device)
        tgt = tgt.to(device)
        
        tgt_input = tgt[:, :-1]
        tgt_labels = tgt[:, 1:]
        
        logits = model(src, tgt_input)
        loss = criterion(
            logits.reshape(-1, logits.size(-1)),
            tgt_labels.reshape(-1)
        )
        
        total_loss += loss.item()
    
    return total_loss / len(dataloader)


# === GENERACIÓN (TRADUCCIÓN) ===

@torch.no_grad()
def translate(model, src_sentence, src_vocab, tgt_vocab, device, max_len=50):
    """
    Traducir una oración.
    
    Esta es la parte mágica: generar palabra por palabra.
    """
    model.eval()
    
    # 1. Convertir entrada a IDs
    src_ids = [src_vocab['<SOS>']]
    for word in src_sentence.lower().split():
        src_ids.append(src_vocab.get(word, src_vocab['<UNK>']))
    src_ids.append(src_vocab['<EOS>'])
    src = torch.tensor([src_ids]).to(device)
    
    # 2. Encodear la entrada
    src_mask = model.create_padding_mask(src)
    encoder_output = model.encode(src, src_mask)
    
    # 3. Generar traducción palabra por palabra
    tgt_ids = [tgt_vocab['<SOS>']]  # Empezar con token especial
    
    for _ in range(max_len):
        # Convertir a tensor
        tgt = torch.tensor([tgt_ids]).to(device)
        tgt_mask = model.create_causal_mask(tgt)
        
        # Decodear
        decoder_output = model.decode(tgt, encoder_output, src_mask, tgt_mask)
        
        # Obtener logits para la última posición
        logits = model.output_projection(decoder_output[:, -1, :])
        
        # Elegir palabra más probable (greedy decoding)
        next_word_id = torch.argmax(logits, dim=-1).item()
        
        # Añadir a la secuencia
        tgt_ids.append(next_word_id)
        
        # Parar si generamos <EOS>
        if next_word_id == tgt_vocab['<EOS>']:
            break
    
    # 4. Convertir IDs de vuelta a palabras
    inv_tgt_vocab = {v: k for k, v in tgt_vocab.items()}
    translation = []
    for id in tgt_ids[1:-1]:  # Excluir <SOS> y <EOS>
        translation.append(inv_tgt_vocab.get(id, '<UNK>'))
    
    return ' '.join(translation)
